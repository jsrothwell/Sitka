interface SkeletonProps {
  width?: string | number;
  height?: string | number;
  rounded?: "none" | "sm" | "md" | "lg" | "full";
  animate?: boolean;
  className?: string;
}

const RADIUS = {
  none: "rounded-none",
  sm:   "rounded",
  md:   "rounded-lg",
  lg:   "rounded-xl",
  full: "rounded-full",
} as const;

export function Skeleton({
  width,
  height,
  rounded = "md",
  animate = true,
  className,
}: SkeletonProps) {
  return (
    <div
      className={cn(
        "bg-[rgb(var(--surface-raised))]",
        RADIUS[rounded],
        animate && "animate-pulse",
        className
      )}
      style={{
        width:  typeof width  === "number" ? `${width}px`  : width,
        height: typeof height === "number" ? `${height}px` : height,
      }}
      aria-hidden="true"
    />
  );
}
