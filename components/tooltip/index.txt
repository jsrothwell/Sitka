struct SitkaTooltip<Content: View>: View {
    let label: String
    var placement: Edge = .top
    @ViewBuilder let content: () -> Content

    @State private var isVisible = false

    var body: some View {
        content()
            .onHover { hovering in
                withAnimation(.easeOut(duration: 0.15)) {

                if isVisible {
